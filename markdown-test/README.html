standaloneComment = \markuplist {
    \line { Letzte Hoffnung D 911/16, T. 5-7}
    \line { Polymetrische Projektion }
}

#(define-public score-type "one-system-example")

% include global configuration file
\include "dissGlobal.ily"


% General configuration for examples in this folder
\include "bsp_conf.ily"

\version "2.17.11"

% Load empty default variables for score block
\include "oll-includes/score-blocks/lied-voice-piano_defaults.ily"

% Allow the staves to have individual time signatures
\layout {
  \context {
    \Score
    \remove "Timing_translator"
    \remove "Default_bar_line_engraver"
  }
  \context {
    \Staff
    \consists "Timing_translator"
    \consists "Default_bar_line_engraver"
    % Don't print time signature, because they are disturbing
    \remove "Time_signature_engraver"
  }
}


% Define entry skip of the example
startSkip = {
  \partial 8 s8 |
  s2.*3 | %1-3
  s2 s8
}

% Define start and end point of the example
skipsAndBreaks = {
  \set Score.skipTypesetting = ##t
  \startSkip
  \set Score.skipTypesetting = ##f
    s8 
    \showNextBarNumber | %4
  s2.*3 | %5-6
  s8
  \set Score.skipTypesetting = ##t
}

% include music contents 
\include "globals.ily"
\include "vocals.ily"
\include "pianoUp.ily"
\include "pianoDown.ily"
\include "pianoDynamics.ily"

% Define time signatures for vocal line ...
vocalsTimesigs = {
  \partial 8
  s8 |
  s2.*4 | % Vorspiel
  \time 4/4
  s2 
  \bar "!" 
  s2
  \time 2/4
  s2
  \time 3/4
  s2 \bar "!"
}

% ... and merge it with the music
vocals = {
  <<
  \vocals
  \vocalsTimesigs
  >>
}

% Define time signatures for the piano ...
pianoTimesigs = {
  \partial 8
  s8
  s2.*4
  \time 1/8 s8
  \time 4/8
  s2*2
  \time 3/8
  s4.
  \time 3/4
}

% ... and merge it with the music
pianoUp = {
  <<
    \pianoUp
    \pianoTimesigs
  >>
}
pianoDown = {
  <<
    \pianoDown
    \pianoTimesigs
  >>
}

% Include actual score block
\compileScore \include "oll-includes/score-blocks/lied-voice-piano.ily"
